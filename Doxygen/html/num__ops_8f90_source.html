366  ! Finds the zero of a function using Zhang's method, which is simpler than
367  ! Brent's method. Taken from from Steven Stage's correction of Zhang's
368  ! paper [http://www.cscjournals.org/manuscript/Journals/IJEA/Volume4/Issue1
369  ! /IJEA-33.pdf].
370  ! Unlike Householder, Zhang's method needs an interval "x_int_in" to work
371  ! in, not a guess. Note that this interval needs to be so that the function
372  ! values at either end are of different value.
373  function calc_zero_zhang(zero,fun,x_int_in) result(err_msg)
374  use num_vars, only: max_it_zero, tol_zero

378  interface
379  function fun(x)
380  use num_vars, only: dp

382  real(dp) :: fun
383  end function fun
384  end interface

389  logical :: converged ! whether converged
390  integer :: jd ! counter
391  integer :: id_loc ! local id (either 1 or 2)
392  real(dp) :: x_int(2) ! points of current interval [a,b]
393  real(dp) :: x_mid ! x at midpoint [c]
394  real(dp) :: x_rule ! x found with rule [s]
395  real(dp) :: x_temp ! temporary x

397  real(dp) :: fun_mid ! function at x_mid

418  zero = x_int(2)

438  ! iterations
439  do jd = 1,max_it_zero
440  ! calculate midpoint and function value
441  x_mid = 0.5_dp*sum(x_int)
442  fun_mid = fun(x_mid)
443 
444  if (abs(fun_int(1)-fun_mid).gt.tol_zero .and. &
445  &abs(fun_int(2)-fun_mid).gt.tol_zero) then
446  ! three different function values: use inverse quadratic
447  ! interpolation
448  x_rule = fun_mid*fun_int(2)*x_int(1)/&
449  &((fun_int(1)-fun_mid)*(fun_int(1)-fun_int(2))) + &
450  &fun_int(1)*fun_int(2)*x_mid/&
451  &((fun_mid-fun_int(1))*(fun_mid-fun_int(2))) + &
452  &fun_int(1)*fun_mid*x_int(2)/&
453  &((fun_int(2)-fun_mid)*(fun_int(2)-fun_mid))
454 
455  if (x_int(1).lt.x_rule .and. x_rule.lt.x_int(2)) then
456  ! found x within interval
457  else
458  ! use bisection instead
459  if (fun_int(1)*fun_mid.lt.0) then ! [a,c] contains sign change
460  id_loc = 1
461  else if (fun_int(2)*fun_mid.lt.0) then ! [b,c] contains sign change
462  id_loc = 2
463  else
464  err_msg = 'Something went wrong...'
465  converged = .true.
466  end if
467  if (.not.converged) x_rule = 0.5_dp*(x_int(id_loc)+x_mid)
468  end if
469  else
470  ! two function values overlap so use secant rule
471  if (fun_int(1)*fun_mid.lt.0) then ! [a,c] contains sign change
472  id_loc = 1
473  else if (fun_int(2)*fun_mid.lt.0) then ! [b,c] contains sign change
474  id_loc = 2
475  else
476  err_msg = 'Something went wrong...'
477  converged = .true.
478  end if
479  if (.not.converged) x_rule = x_int(id_loc) - &
480  &fun_int(id_loc)*(x_int(id_loc)-x_mid)/&
481  &(fun_int(id_loc)-fun_mid)
482  end if
483 
484  ! calculate fun(s)
485  fun_rule = fun(x_rule)
486 
487  ! make sure x_mid < x_rule
488  if (x_mid.gt.x_rule) then
489  x_temp = x_rule
490  x_rule = x_mid
491  x_mid = x_temp
492  end if
493 
494  ! check whether [c,s] contains root
495  if (fun_mid*fun_rule.lt.0) then
496  x_int(1) = x_mid
497  x_int(2) = x_rule
498  else
499  if (fun_int(1)*fun_mid.lt.0) then
500  x_int(2) = x_mid
501  else
502  x_int(1) = x_rule
503  end if
504  end if
505 
506 #if ldebug
507  if (debug_calc_zero) then
508  x_ints(jd,:) = x_int
509  end if
510 #endif
511 
512  ! check for convergence
513  fun_int = [fun(x_int(1)),fun(x_int(2))]
514  if (abs(fun_int(1)).lt.tol_zero) then
515  zero = x_int(1)
516  converged = .true.
517  end if
518  if (abs(fun_int(2)).lt.tol_zero) then
519  zero = x_int(2)
520  converged = .true.
521  end if
522  if ((x_int(2)-x_int(1)).lt.tol_zero) then
523  zero = 0.5_dp*sum(x_int)
524  converged = .true.
525  end if
526 
527  if (converged) then
528 #if ldebug
529  if (debug_calc_zero) call plot_evolution(x_ints(1:jd,:))
530 #endif
531  return
532  end if
533  end do
