109  ! setup parallel id:
110  ! par_id(1): start index
111  ! par_id(2): end index
112  ! par_id(3): stride
113  ! These are set up by considering the transformation between a point r
114  ! (1..n) in the local variable, with corresponding indices (a..b) indicated
115  ! by par_lim in the HDF5 variable in memory, so that n = b-a+1:
116  ! p + k s = a + r - 1,
117  ! where k is an integer, s is the stride for Richardson level i (with max.
118  ! level I):
119  ! s = 2^(I-1) for i = 1
120  ! = 2^(I-i+1) for i > 1
121  ! and where p is the starting point in the HDF5 variables:
122  ! p = 1 for i = 1
123  ! = 1 + s/2 for i > 1
124  ! Therefore, the minimum possible r lies in (0..s-1):
125  ! mod(r-1,s) = r_min-1,
126  ! which leads to
127  ! r_min = 1 + mod(p-a,s).
128  ! The maximum possible r, then, lies in (n-s+1..n), which leads to:
129  ! r_max = n - s + 1 + mod(p-b-1,s).
130  ! These limits and the strides are saved in par_id.
131  ! If the optional indices a and b are not given they are assumed to be 1 and
132  ! n, with n = 1+ks, which simplifies the equations to:
133  ! r_min = p
134  ! r_max = n-p+1.
135  ! Optionally, the indices in the HDF5 arrays can also be returned. These are
136  ! equal to k-1, where k is the integer refered to above:
137  ! par_id_mem = 1 + (a-1-p+par_id)/s.
138  function setup_par_id(grid,rich_lvl_max,rich_lvl_loc,tot_rich,par_lim,&
139  &par_id_mem) result(par_id)
140  use grid_vars, only: grid_type
141 
142  ! input / output
143  type(grid_type), intent(in) :: grid ! grid
144  integer, intent(in) :: rich_lvl_max ! maximum Richardson level
145  integer, intent(in) :: rich_lvl_loc ! local Richardson level
146  logical, intent(in), optional :: tot_rich ! whether to combine with previous Richardson levels
147  integer, intent(in), optional :: par_lim(2) ! limits (a..b) of variable requested
148  integer, intent(inout), optional :: par_id_mem(2) ! parallel id in memory
149  integer :: par_id(3) ! parallel id
150 
151  ! local variables
152  integer :: s ! stride
153  integer :: p ! lower limit in memory
154  integer :: n ! b-a+1
155  logical :: tot_rich_loc ! local tot_rich
156  integer :: par_lim_loc(2) ! local par_lim
157 
158  ! set up local tot_rich and par_lim
159  tot_rich_loc = .false.
160  if (present(tot_rich) .and. rich_lvl_max.gt.1) tot_rich_loc = tot_rich ! only for higher Richardson levels
161  par_lim_loc = [1,grid%n(1)]
162  if (present(par_lim)) par_lim_loc = par_lim
163 
164  ! set up parallel indices
165  n = par_lim_loc(2)-par_lim_loc(1)+1
166  if (tot_rich_loc) then
167  if (rich_lvl_loc.eq.1) then
168  s = 2**(rich_lvl_max-1)
169  p = 1
170  else
171  s = 2**(rich_lvl_max+1-rich_lvl_loc)
172  p = 1 + s/2
173  end if
174  else
175  s = 1
176  p = 1
177  end if
178  par_id(1) = 1 + modulo(p-par_lim_loc(1),s)
179  par_id(2) = n - s + 1 + modulo(p-par_lim_loc(2)-1,s)
180  par_id(3) = s
181 
182  ! set par_id in memory if requested
183  if (present(par_id_mem)) par_id_mem = &
184  &1 + (par_lim_loc(1)-1-p+par_id(1:2))/s
185  end function setup_par_id
