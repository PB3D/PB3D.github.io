4 ! Splines are fully specified by the interpolation points, except that at !
5 ! the ends, we have the freedom to prescribe the second derivatives. If we !
6 ! know a derivative at an end (exactly), then best is to impose that. !
7 ! Otherwise, it is better to use the "consistent" end conditions: the second !
8 ! derivative is determined such that it is smooth. !
9 ! !
10 ! This module is based on a code written by John E. Pask, LLNL. !
11 !------------------------------------------------------------------------------!

26  ! Takes the function values 'y' on the grid 'x' and returns new values
27  ! 'ynew' at the given grid 'xnew' using cubic splines interpolation with
28  ! such boundary conditions so that the 2nd derivative is consistent with the
29  ! interpolating cubic.
30  integer function spline3_real(x, y, xnew, ynew, dynew, d2ynew) result(ierr) ! real version

34  real(dp), intent(in) :: x(:), y(:), xnew(:)
35  real(dp), intent(out), optional :: ynew(:), dynew(:), d2ynew(:)
36

38  real(dp) :: c(0:4, size(x)-1)
39  integer :: i, i2
40  integer :: i_min
41  ! initialize ierr
42  ierr = 0
43 
44  ! set up spline coefficients
45  ierr = spline3pars(x, y, [2, 2], [0._dp, 0._dp], c)
46  chckerr('')
47 
48  i2 = 0
49  do i = 1, size(xnew)
50  i_min = i2
51  ierr = iixmin(xnew(i), x, i_min,i2)
52  chckerr('')
53  if (present( ynew)) ynew(i) = poly3(xnew(i), c(:, i2))
54  if (present( dynew)) dynew(i) = dpoly3(xnew(i), c(:, i2))

107  ! Returns parameters c defining cubic spline interpolating x-y data xi, yi,
108  ! with boundary conditions specified by bcytpe:
109  ! - bctype(1) = type at left end, bctype(2) = type at right end.
110  ! - 1 = specified 2nd derivative, 2 = 2nd derivative consistent with
111  ! interpolating cubic,
112  ! and bcvals:
113  ! - bcval(1) = value at left end, bcval(2) = value at right end.
114  ! The output parameters that define the spline are c(i,j), where
115  ! c(i,j) = ith parameter of jth spline polynomial,
116  ! p_j = sum_{i=1}^4 c(i,j) (x-c(0,j))^(i-1), j = 1..n-1,
117  ! n = # of data pts.
118  ! The dimensions of c are: c(0:4,1:n-1).
119  integer function spline3pars(xi,yi,bctype,bcval,c) result(ierr)
120  character(*), parameter :: rout_name = 'spline3pars'
121 
122  ! input / ouptut
123  real(dp), intent(in) :: xi(:) ! x values of data
124  real(dp), intent(in) :: yi(:) ! y values of data
125  integer, intent(in) :: bctype(2) ! type of boundary condition at each end:
126  real(dp), intent(in) :: bcval(2) ! boundary condition values at each end:
127  real(dp), intent(out) :: c(0:,:) ! parameters defining spline
128 
129  ! local variables
130  real(dp) :: As(5,2*size(c,2)) ! spline eq. matrix -- LAPACK band form
131  real(dp) :: bs(2*size(c,2)) ! spline eq. rhs vector
132  real(dp) :: cs(2*size(c,2)) ! spline eq. solution vector
133  real(dp) :: hi(size(c,2)) ! spline intervals
134  real(dp) :: Ae(4,4) ! end-cubic eq. matrix
135  real(dp) :: be(4) ! end-cubic eq. rhs vector
136  real(dp) :: ce(4) ! end-cubic eq. solution vector
137  real(dp) :: xe(4),ye(4) ! x,y values at ends
138  real(dp) :: d2p1,d2pn ! 2nd derivatives at ends
139  real(dp) :: x0 ! expansion center
140  real(dp) :: c1,c2,c3,c4 ! expansion coefficients
141  integer :: n ! number of data points
142  integer :: i,j,i2 ! counters
143 
144  ! local lapack variables
145  integer :: ipiv(4),ipiv2(2*size(c,2))
146  real(dp) :: bemat(4,1),bmat(2*size(c,2),1)
147  integer :: info
148 
149  ! initialize ierr
150  ierr = 0
151 
152  ! check input parameters
153  if (bctype(1) < 1 .or. bctype(1) > 2) ierr = 1
154  chckerr('bctype /= 1 or 2.')
155  if (bctype(2) < 1 .or. bctype(2) > 2) ierr = 1
156  chckerr('bctype /= 1 or 2.')
157  if (size(c,1) /= 5) ierr = 1
158  chckerr('size(c,1) /= 5.')
159  if (size(c,2) /= size(xi)-1) ierr = 1
160  chckerr('size(c,2) /= size(xi)-1.')
161  if (size(xi) /= size(yi)) ierr = 1
162  chckerr('size(xi) /= size(yi)')
163 
164  ! To get rid of compiler warnings:
165  d2p1 = 0
166  d2pn = 0
167 
168  ! initializations
169  n=size(xi)
170  do i=1,n-1
171  hi(i)=xi(i+1)-xi(i)
172  end do
173 
174  ! compute interpolating-cubic 2nd derivs at ends, if required
175  ! left end
176  if(bctype(1)==2) then
177  if (n < 4) ierr = 1
178  chckerr('n < 4')
179  xe=xi(1:4)
180  ye=yi(1:4)
181  x0=xe(1) ! center at end
182  do i=1,4
183  do j=1,4
184  ae(i,j) = (xe(i)-x0)**(j-1)
185  end do
186  end do
187  ae(:,1) = 1 ! set 0^0 = 1
188  be=ye; bemat(:,1)=be
189  call dgesv(4, 1, ae, 4, ipiv, bemat, 4, info)
190  if (info /= 0) ierr = 1
191  chckerr('dgesv error')
192  ce=bemat(:,1)
193  d2p1=2*ce(3)
194  else
195  d2p1=bcval(1)
196  end if
197 
198  ! right end
199  if(bctype(2)==2) then
200  if (n < 4) ierr = 1
201  chckerr('n < 4')
202  xe=xi(n-3:n)
203  ye=yi(n-3:n)
204  x0=xe(4) ! center at end
205  do i=1,4
206  do j=1,4
207  ae(i,j) = (xe(i)-x0)**(j-1)
208  end do
209  end do
210  ae(:,1) = 1 ! set 0^0 = 1
211  be=ye; bemat(:,1)=be
212  call dgesv(4, 1, ae, 4, ipiv, bemat, 4, info)
213  if (info /= 0) ierr = 1
214  chckerr('dgesv error')
215  ce=bemat(:,1)
216  d2pn=2*ce(3)
217  else
218  d2pn=bcval(2)
219  end if
220 
221  ! construct spline equations -- LAPACK band form
222  ! basis:
223  ! - phi1 = -(x-x_i)/h_i,
224  ! - phi2 = x-x_{i+1})/h_i,
225  ! - phi3 = phi1^3-phi1,
226  ! - phi4 = phi2^3-phi2,
227  ! on interval [x_i,x_{i+1}] of length h_i = x_{i+1}-x_i
228  as=0
229 
230  ! left end condition
231  as(4,1) = 6/hi(1)**2
232  bs(1) = d2p1
233 
234  ! internal knot conditions
235  do i=2,n-1
236  i2=2*(i-1)
237  as(5,i2-1) = 1/hi(i-1)
238  as(4,i2) = 2/hi(i-1)
239  as(3,i2+1) = 2/hi(i)
240  as(2,i2+2) = 1/hi(i)
241  as(5,i2) = 1/hi(i-1)**2
242  as(4,i2+1) = -1/hi(i)**2
243  bs(i2) = (yi(i+1) - yi(i))/hi(i) - (yi(i) - yi(i-1))/hi(i-1)
244  bs(i2+1) = 0
245  end do
246 
247  ! right end condition
248  as(4,2*(n-1)) = 6/hi(n-1)**2
249  bs(2*(n-1)) = d2pn
250 
251  ! solve spline equations -- LAPACK band form
252  bmat(:,1)=bs
253  call dgbsv(2*(n-1),1,2,1,as,5,ipiv2,bmat,2*(n-1),info)
254  if (info /= 0) ierr = 1
255  chckerr('dgbsv error')
256  cs=bmat(:,1)
257  !write(*,*) cs(1:6)
258  !write(*,*) cs(2*(n-1)-5:2*(n-1))
259 
260  ! transform to (x-x0)^(i-1) basis and return
261  do i=1,n-1
262  ! coefficients in spline basis:
263  c1=yi(i)
264  c2=yi(i+1)
265  c3=cs(2*i-1)
266  c4=cs(2*i)
267 
268  ! coefficients in (x-x0)^(i-1) basis
269  c(0,i)=xi(i)
270  c(1,i)=c1
271  c(2,i)=-(c1-c2+2*c3+c4)/hi(i)
272  c(3,i)=3*c3/hi(i)**2
273  c(4,i)=(-c3+c4)/hi(i)**3
274  end do
275  end function spline3pars

277  ! Returns index i of interval [xi(i),xi(i+1)] containing x in mesh xi,
278  ! with intervals indexed by left-most points.
279  ! N.B.: x outside [x1,xn] are indexed to nearest end.
280  ! Uses bisection, except if "x" lies in the first or second elements (which
281  ! is often the case)
282  integer function iix(x,xi,i1) result(ierr)
283  character(*), parameter :: rout_name = 'iix'
284 
285  ! input / output
286  real(dp), intent(in) :: x ! target value
287  real(dp), intent(in) :: xi(:) ! mesh, xi(i) < xi(i+1)
288  integer :: i1 ! index
289 
290  ! local variables
291  integer n ! number of mesh points
292  integer i2, ic
293 
294  ! initialize ierr
295  ierr = 0
296 
297  n = size(xi)
298  i1 = 1
299  if (n < 2) then
300  ierr = 1
301  chckerr('n < 2')
302  elseif (n == 2) then
303  i1 = 1
304  elseif (n == 3) then
305  if (x <= xi(2)) then ! first element
306  i1 = 1
307  else
308  i1 = 2
309  end if
310  elseif (x <= xi(1)) then ! left end
311  i1 = 1
312  elseif (x <= xi(2)) then ! first element
313  i1 = 1
314  elseif (x <= xi(3)) then ! second element
315  i1 = 2
316  elseif (x >= xi(n)) then ! right end
317  i1 = n-1
318  else
319  ! bisection: xi(i1) <= x < xi(i2)
320  i1 = 3; i2 = n
321  do
322  if (i2 - i1 == 1) exit
323  ic = i1 + (i2 - i1)/2
324  if (x >= xi(ic)) then
325  i1 = ic
326  else
327  i2 = ic
328  endif
329  end do
330  end if
331  end function iix
