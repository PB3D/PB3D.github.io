272 
273  ! input / output
274  real(dp), intent(in) :: var(:) ! parallel vector
275  real(dp), intent(inout) :: dis_var(:) ! redistributed vector
276  integer, intent(in) :: lims(2) ! indices of parallel vector
277  integer, intent(in) :: lims_dis(2) ! indices of redistributed parallel vector
278 
279  ! local variables
280  integer :: id, jd ! counters
281  integer, allocatable :: lims_tot(:,:) ! limits of all processes
282  integer, allocatable :: lims_dis_tot(:,:) ! redistributed limits of all processes
283  integer, allocatable :: temp_lim(:) ! temporary variable
284  integer, allocatable :: n_vars(:,:) ! number of values to be received and sent from each process (row: to, column: from)
285  integer, allocatable :: nr_sen(:) ! number of values to be sent to each process from this process
286  integer, allocatable :: nr_rec(:) ! number of values to be received from each process to this process
287  integer, allocatable :: id_rec(:) ! starting position of variable to be received in local memory (starting at 0)
288  integer, allocatable :: id_sen(:) ! starting position of variable to be sent in local memory (starting at 0)
289  integer :: max_loc ! local maximum
290 
291  ! initialize ierr
292  ierr = 0
293 
294  ! get limits and redistributed limits of all procs
295  allocate(lims_tot(2,n_procs))
296  allocate(lims_dis_tot(2,n_procs))
297  do id = 1,2
298  ierr = get_ser_var(lims(id:id),temp_lim,scatter=.true.)
299  chckerr('')
300  lims_tot(id,:) = temp_lim
301  ierr = get_ser_var(lims_dis(id:id),temp_lim,scatter=.true.)
302  chckerr('')
303  lims_dis_tot(id,:) = temp_lim
304  end do
305 
306  ! find out how many variables to send and receive. A row indicates a
307  ! receive and a column a send. E.g.:
308  ! (a b)
309  ! (c d)
310  ! means that process 0 gets a from process 0 and b from process 1.
311  ! Process 1 gets c from process 0 and d from process 1.
312  allocate(n_vars(n_procs,n_procs))
313  n_vars = 0
314  do jd = 1,n_procs
315  max_loc = lims_dis_tot(1,jd)
316  do id = 1,n_procs
317  if (lims_tot(2,id).ge.max_loc) then
318  n_vars(jd,id) = min(lims_tot(2,id),lims_dis_tot(2,jd)) - &
319  &max_loc + 1
320  max_loc = max_loc + n_vars(jd,id)
321  end if
322  end do
323  end do
324 
325  ! set up how many are to be sent by each process and which memory index
326  ! - nr_rec is the row of n_vars corresponding to this process,
327  ! - nr_sen is the column of n_vars corresponding to this process,
328  ! - id_rec counts the number of variables gotten from previous
329  ! processes,
330  ! - id_sen counts the number of variables sent to previous processes,
331  ! taking into account the difference betweeen the lower limit of the
332  ! process sending to and the current process.
333  allocate(nr_rec(n_procs))
334  allocate(nr_sen(n_procs))
335  allocate(id_rec(n_procs))
336  allocate(id_sen(n_procs))
337  do id = 1,n_procs
338  nr_rec(id) = n_vars(rank+1,id)
339  nr_sen(id) = n_vars(id,rank+1)
340  id_rec(id) = sum(n_vars(rank+1,1:id-1))
341  id_sen(id) = sum(n_vars(id,1:rank)) + lims_dis_tot(1,id) - lims(1)
342  end do
343 
344  call mpi_alltoallv(var,nr_sen,id_sen,mpi_double_precision,&
345  &dis_var,nr_rec,id_rec,mpi_double_precision,mpi_comm_world,ierr)
346  chckerr('')

359  ! Fill the ghost regions in an array by sending the first normal point of a
360  ! process to the left process. Every message is identified by its sending
361  ! process. The array should have the extended size, including ghost regions.
362  ! [MPI] Collective call
363  integer function get_ghost_arr_3d_complex(arr,size_ghost) result(ierr) ! 3D complex version
364  use num_vars, only: rank, n_procs
365 
366  character(*), parameter :: rout_name = 'get_ghost_arr_3D_complex'
367 
368  ! input / output
369  complex(dp), intent(inout) :: arr(:,:,:) ! divided array
370  integer, intent(in) :: size_ghost ! width of ghost region
371 
372  ! local variables
373  integer :: n_modes(2) ! number of modes
374  integer :: tot_size ! total size (including ghost region)
375  integer :: istat(mpi_status_size) ! status of send-receive
376 
377  ! initialize ierr
378  ierr = 0
379 
380  ! initialize number of modes and total size
381  n_modes = [size(arr,1),size(arr,2)]
382  tot_size = size(arr,3)
383 
384  ! ghost regions only make sense if there is more than 1 process
385  if (n_procs.gt.1) then
386  if (rank.eq.0) then ! first rank only receives
387  call mpi_recv(arr(:,:,tot_size-size_ghost+1:tot_size),&
388  &size_ghost*product(n_modes),mpi_double_complex,rank+1,&
389  &rank+1,mpi_comm_world,istat,ierr)
390  chckerr('Failed to receive')
391  else if (rank+1.eq.n_procs) then ! last rank only sends
392  call mpi_send(arr(:,:,1:size_ghost),&
393  &size_ghost*product(n_modes),mpi_double_complex,rank-1,&
394  &rank,mpi_comm_world,ierr)
395  chckerr('Failed to send')
396  else ! middle ranks send and receive
397  call mpi_sendrecv(arr(:,:,1:size_ghost),&
398  &size_ghost*product(n_modes),mpi_double_complex,rank-1,&
399  &rank,arr(:,:,tot_size-size_ghost+1:tot_size),&
400  &size_ghost*product(n_modes),mpi_double_complex,&
401  &rank+1,rank+1,mpi_comm_world,istat,ierr)
402  chckerr('Failed to send and receive')
403  end if
404  end if
405  end function get_ghost_arr_3d_complex

612  integer function lock_req_acc(lock,blocking) result(ierr)

617

619  type(lock_type), intent(inout) :: lock ! lock

631 #endif

642 #endif

650  &'requesting access'

673 #if ldebug

682  allocate(ranks_to_activate(size(next_nb_procs)+1))

708  chckerr('')

803  if (next_proc_exists) then
804  do id = 1,size(ranks_to_activate)

806  chckerr('')
807  end do
808  end if
809  end if

812  ! Decides whether a waiting list is empty.
813  ! The type of process to find is indicated by an array of possible values.
814  ! See lock_wl_change for an explanation of the process type.
815  ! Additionally, for NB processes, the negative inverse of these values are
816  ! used.
817  ! If the waiting list is not empty, the next process(es) can optionally be
818  ! returned.
819  logical function wl_empty(wl,proc_type,next_procs)
820  use num_vars, only: n_procs, rank
821 
822  ! input / output
823  integer, intent(in) :: wl(:) ! waiting list
824  integer, intent(in) :: proc_type(:) ! types of processes accepted
825  integer, intent(inout), optional, allocatable :: next_procs(:) ! next process(es) if not empty
826

832  ! initialize
833  nr_np = 0

839  ! wrap around n_procs-1 to zero
840  next_procs_loc(nr_np+1) = mod(rank+id,n_procs)

844  wl_empty = .false.
845  nr_np = nr_np + 1
846  exit proc_types
847  end if
848  end do proc_types
849  end do
850 
851  if (.not.wl_empty .and. present(next_procs)) then
852  if (allocated(next_procs)) deallocate(next_procs)
853  allocate(next_procs(nr_np))
854  next_procs = next_procs_loc(1:nr_np)
855  end if

916  ! Adds, removes or sets to active a rank from the waiting list for a lock
917  ! and returns the lock waiting list:
918  ! wl_action = 0: remove
919  ! wl_action = 1: add
920  ! wl_action = 2: active
921  ! Or negative equivalents for NB procs.
922  ! Optionally, the rank(s) of the process for which to perform this action
923  ! can be passed. This is useful for doing the same action on multiple
924  ! processes, as is described in the note of lock_req_acc.
925  ! (based on http://www.mcs.anl.gov/~thakur/papers/atomic-mode.pdf)
926  integer function lock_wl_change(wl_action,blocking,lock,wl,ranks) &
927  &result(ierr)
928  use num_vars, only: n_procs, rank
929 
930  character(*), parameter :: rout_name = 'lock_wl_change'
931 
932  ! input / output
933  integer, intent(in) :: wl_action ! action to perform
934  logical, intent(in) :: blocking ! the ranks to be changed are blocking
935  type(lock_type), intent(inout) :: lock ! lock
936  integer, intent(inout), allocatable :: wl(:) ! waiting list
937  integer, intent(in), optional :: ranks(:) ! rank(s) for which to perform option
938 
939  ! local variables
940  integer :: id ! counter
941  integer :: n_ranks ! number of ranks to set
942  integer :: put_val ! value to put
943  integer, allocatable :: ranks_loc(:) ! local ranks
944  integer(kind=MPI_ADDRESS_KIND) :: one = 1 ! one
945  integer(kind=MPI_ADDRESS_KIND) :: disp ! local displacement
946  integer :: ln ! local length
947 #if ldebug
948  integer :: istat ! status
949  integer(kind=8) :: window_time(2) ! time that a window is locked
950 #endif
951 
952  ! initialize ierr
953  ierr = 0
954 
955  ! set value to put
956  put_val = wl_action
957  if (.not.blocking) put_val = -put_val ! NB indicated in waiting list using a negative value
958  !allocate(wl_excl(n_procs-1))
959  if (allocated(wl)) deallocate(wl)
960  allocate(wl(n_procs))
961 
962  ! set local ranks if not present
963  if (.not.present(ranks)) then
964  allocate(ranks_loc(1))
965  ranks_loc(1) = rank
966  else
967  allocate(ranks_loc(size(ranks)))
968  ranks_loc = ranks
969  end if
970  n_ranks = size(ranks_loc)
971 
972  ! get window to lock
973 #if ldebug
974  if (debug_lock) call system_clock(window_time(1))
975 #endif
976  call mpi_win_lock(mpi_lock_exclusive,0,0,lock%wl_win,ierr)
977  chckerr('Failed to lock window')
978 
979  ! get previous list and put value for current ranks
980  ! Do this by looking at every rank that is to be put, whether there are
981  ! values to be gotten between the previous put and the current put.
982  ! Finally, do one more get for the ranks between the last put and the
983  ! end.
984  do id = 1,n_ranks+1
985  ! set local displacement and length
986  if (id.eq.1) then
987  disp = 0
988  ln = ranks_loc(1)
989  else
990  disp = disp + ln + 1 ! previous step used ln gets and 1 put
991  if (id.lt.n_ranks+1) then
992  ln = ranks_loc(id)-ranks_loc(id-1)-1
993  else
994  ln = n_procs-ranks_loc(id-1)-1
995  end if
996  end if
997 
998  ! perform the gets between this rank and previous rank
999  if (ln.gt.0) then ! if there is something to get between last put and this one
1000  call mpi_get(wl(int(disp)+1:int(disp)+ln),ln,mpi_integer,0,&
1001  &disp,ln,mpi_integer,lock%wl_win,ierr)
1002  chckerr('Failed to get waiting list')
1003  end if
1004 
1005  ! perform the single put up to the current rank if not last piece
1006  if (id.lt.n_ranks+1) then
1007  call mpi_put(put_val,1,mpi_integer,0,ranks_loc(id)*one,1,&
1008  &mpi_integer,lock%wl_win,ierr)
1009  chckerr('Failed to add to waiting list')
1010  wl(ranks_loc(id)+1) = put_val
1011  end if
1012  end do
1013 
1014  ! unlock window
1015  call mpi_win_unlock(0,lock%wl_win,ierr)
1016  chckerr('Failed to lock window')
1017 #if ldebug
1018  if (debug_lock) call system_clock(window_time(2))
1019 #endif
1020 
1021 #if ldebug
1022  if(debug_lock) write(*,*,iostat=istat) trim(lock_header(lock)),&
1023  &' time: '//&
1024  &trim(r2strt(1.e-9_dp*(window_time(2)-window_time(1)))),&
1025  &' waiting list:', wl
1026 #endif
1027  end function lock_wl_change
