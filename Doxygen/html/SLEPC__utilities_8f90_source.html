34  ! Insert a block pertaining to 1 normal point into a matrix. Optionally,
35  ! also set the Hermitian transpose and / or overwrite instead of add value.
36  ! This routine takes into account that if the mode numbers change as a
37  ! function of the normal coordinate, the local blocks, which were set up
38  ! without taking this into account, have to be shifted. The therefore
39  ! missing information is omitted, and the corresponding matrix elements are
40  ! set to zero.
41  ! This is justifiable if there are many more mode couplings that are not
42  ! omitted (i.e. when n_mod_X is large), so that it is a small effect,
43  ! and/or when the coupling between these modes is already negligible.
44  ! Unfortunately, it can be argued that this is not generally the case for
45  ! the fast version (X_type = 2) since PV^1 and PV^2 are both ~ (nq-m), which
46  ! indicates that the smallest elements in the corresponding off-diagonal
47  ! blocks lie in the central columns for PV^1 and in the central columns and
48  ! rows for PV^2, while the problematic elements lie farthest away from the
49  ! central columns for PV^1 and the central columns and rows for PV^2, so
50  ! these are generally not small.
51  ! The coefficients due to normal discretization get increasingly small
52  ! for higher orders, though, which compensates this somewhat for higher
53  ! orders. Finally, an alternative would be to set the problematic elements
54  ! not to zero but equal to the closest element, but then consistency
55  ! would be sacrified, for example in the energy reconstruction of the
56  ! POST-processing: In the energy reconstruction the exact same terms are
57  ! neglected as well.
58  ! To conclude, for fast X type 2, the number of modes (n_mod_X) has to be
59  ! chosen high enough compared with the variation of the mode numbers; i.e.
60  ! the variation of the safety factor or rotational transform.
61  integer function insert_block_mat(block,mat,r_id,ind,n_sol,transp,&
62  &overwrite) result(ierr)
63  use x_vars, only: n_x, m_x, n_mod_x
64  use num_vars, only: use_pol_flux_f
65 #if ldebug
66  use num_vars, only: rank
67  use input_utilities, only: pause_prog
68 #endif

112  &persistent=.true.)
113  call lvl_ud(1)
114  end if

117  ! only set values if within matrix range

119  ! set error message

122  ! set up local block, possibly translated for fast PB3D
123  allocate(block_loc(size(block,1),size(block,2)))
124  block_loc = 0._dp
125  do m = 1,n_mod_x
126  do k = 1,n_mod_x
127  if (use_pol_flux_f) then
128  k_loc = k + m_x(r_id+1,k) - m_x(r_id+1+ind(1),k)
129  m_loc = m + m_x(r_id+1,k) - m_x(r_id+1+ind(2),k)
130  else
131  k_loc = k + n_x(r_id+1,k) - n_x(r_id+1+ind(1),k)
132  m_loc = m + n_x(r_id+1,k) - n_x(r_id+1+ind(2),k)
133  end if

135  &k_loc.le.n_mod_x .and. m_loc.le.n_mod_x) &
136  block_loc(k_loc,m_loc) = block(k,m)
137  end do
138  end do

180  ! clean up

183 #if ldebug
184  if (debug_insert_block_mat) &
185  &call writo('Due to out of range no local block is added',&
186  &persistent=.true.)
187 #endif
188  end if

195 #endif
196  end function insert_block_mat
