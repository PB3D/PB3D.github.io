34  ! Calculates the normal or geodesic components of the plasma perturbation or
35  ! the magnetic perturbation, indicated by the variable "XUQ_style":
36  ! - XUQ_style = 1: X (supports parallel derivative)
37  ! - XUQ_style = 2: U (supports parallel derivative)
38  ! - XUQ_style = 3: Qn
39  ! - XUQ_style = 4: Qg
40  ! X and U support the calculation of the parallel derivative and for Qn and
41  ! Qg, the metric variables have to be provided as well.
42  ! As this operation requires the equilibrium, metric, perturbation
43  ! and solution variables, the grid in which this happens is not so
44  ! straightforward:
45  ! The equilibrium and metric variables are tabulated in an equilibrium grid,
46  ! which needs to have the same angular extent as the perturbation grid, in
47  ! which the perturbation variables are tabulated. The normal extent of these
48  ! two grids can be different, though, as well as the normal extent of the
49  ! solution grid, in which the solution variables are tabulated.
50  ! Currently, the perturbation variables are supposed to have the same
51  ! angular extent as the equilibrium and metric variables, and the same
52  ! normal extent as the solution variables. The output then follows the
53  ! perturbation grid.
54  integer function calc_xuq_arr(grid_eq,grid_X,eq_1,eq_2,X,sol,X_id,&

136  ! allocate helper variables
137  ! Multiplicative factors fac_0 and fac_1 and par_fac are used in
138  ! XUQ = fac_0*X + fac_1*DX with fac_0 and fac_1:
139  ! 1. fac_0 = 1 (i(nq-m) or i(n-iota m) for deriv)
140  ! fac_1 = 0

152  allocate(xuq_loc(size(xuq,1),size(xuq,2)))
153  allocate(fac_0(grid_x%n(1),grid_x%n(2),grid_x%loc_n_r))
154  allocate(fac_1(grid_x%n(1),grid_x%n(2),grid_x%loc_n_r))
155  allocate(par_fac(grid_x%loc_n_r))
156  allocate(jq(grid_x%loc_n_r))
157  allocate(s(grid_x%n(1),grid_x%n(2),grid_x%loc_n_r))
158  if ((xuq_style.eq.3 .or. xuq_style.eq.4)) &
159  &allocate(inv_j(grid_x%n(1),grid_x%n(2),grid_x%loc_n_r))
160  allocate(expon(grid_x%n(1),grid_x%n(2),grid_x%loc_n_r))

257  ! iterate over all Fourier modes
258  fourier: do ld = 1,n_mod_loc
259  ! set up parallel multiplicative factor par_fac in normal eq grid
260  if (use_pol_flux_f) then
261  par_fac = sol%n(:,ld)*jq-sol%m(:,ld)
262  else
263  par_fac = sol%n(:,ld)-sol%m(:,ld)*jq
264  end if

267  do kd = 1,grid_x%loc_n_r
268  expon(:,:,kd) = sol%n(kd,ld)*grid_x%zeta_F(:,:,kd) - &

272  ! set up multiplicative factors in eq grid
273  select case (xuq_style)
274  case (1) ! calculating X
275  if (deriv_loc) then ! parallel derivative
276  do kd = 1,grid_x%loc_n_r
277  fac_0(:,:,kd) = iu*par_fac(kd)
278  end do
279  else

281  end if
282  fac_1 = 0._dp
283  case (2) ! calculating U
284  if (deriv_loc) then ! parallel derivative
285  fac_0 = x%DU_0(:,:,:,ld)
286  fac_1 = x%DU_1(:,:,:,ld)
287  else
288  fac_0 = x%U_0(:,:,:,ld)
289  fac_1 = x%U_1(:,:,:,ld)
290  end if
291  case (3) ! calculating Qn
292  if (deriv_loc) then ! parallel derivative
293  ierr = 1
294  err_msg = 'For XUQ_style = '//trim(i2str(xuq_style))//&
295  &' calculation of parallel derivative not supported'
296  chckerr('')
297  else
298  do kd = 1,grid_x%loc_n_r
299  fac_0(:,:,kd) = iu*par_fac(kd)*inv_j(:,:,kd)
300  end do
301  end if
302  fac_1 = 0._dp
303  case (4) ! calculating Qg
304  if (deriv_loc) then ! parallel derivative
305  ierr = 1
306  err_msg = 'For XUQ_style = '//trim(i2str(xuq_style))//&
307  &' calculation of parallel derivative not supported'
308  chckerr('')
309  else
310  do kd = 1,grid_x%loc_n_r
311  fac_0(:,:,kd) = x%DU_0(:,:,kd,ld)*inv_j(:,:,kd) &
312  &- s(:,:,kd)
313  fac_1(:,:,kd) = x%DU_1(:,:,kd,ld)*inv_j(:,:,kd)
314  end do
315  end if
316  case default
317  err_msg = 'No XUQ style associated with '//&
318  &trim(i2str(xuq_style))
319  ierr = 1
320  chckerr(err_msg)
321  end select
322 
323  ! iterate over all normal points in sol grid
324  normal: do kd = 1,grid_x%loc_n_r
325  ! set up loc complex XUQ without time at this normal point
326  xuq_loc = exp(iu*expon(:,:,kd)) * &
327  &(sol%vec(ld,kd,x_id)*fac_0(:,:,kd) + &
328  &dsol_vec(ld,kd)*fac_1(:,:,kd))
329 
330  ! iterate over time steps
331  do id = 1,n_t
332  ! add current mode
333  xuq(:,:,kd,id) = xuq(:,:,kd,id) + xuq_loc * &
334  &exp(iu*sqrt_sol_val_norm*time(id)*2*pi)
335  end do
336  end do normal
337  end do fourier
338  end function calc_xuq_arr

374  ! Calculate the total version of the solution vector from the local version
375  ! or the other way around. This is the solution vector for all of the
376  ! possible mode numbers, which can be different from the local mode numbers
377  ! for X style 2:
378  ! - local: the size of the saved perturbation and solution variables,
379  ! prescribed by the user as input--either directly through n_mod_X or
380  ! through the limits on the modes using min_sec_X and max_sec_X.
381  ! - total: all the possible modes that can resonate in the plasma, which
382  ! can be different from the local number for X style 2, as the mode
383  ! numbers depend on the normal coordinate.
384  ! If the output variable is not allocated, it is done here.
385  ! Note: If the lowest limits of the grid is not 1 (e.g. grid_X%i_min = 1 for
386  ! first process), the input variable i_min should be set to set correctly.
387  ! For a full grid, it should be set to 1.
388  integer function calc_tot_sol_vec(i_min,sol_vec_loc,sol_vec_tot) &
389  &result(ierr)

398  complex(dp), intent(inout), allocatable :: sol_vec_tot(:,:) ! solution vector for all possible resonating modes
399 
400  ! local variables
401  character(len=max_str_ln) :: err_msg ! error message
402  integer :: n_mod_loc ! local number of modes that are used in the calculations
403  integer :: n_mod_tot ! total number of modes
404  integer :: kd, ld ! counters
405  integer :: kd_loc ! local kd
406  integer :: n_r ! normal size of variables
407 
408  ! set n_r
409  n_r = size(sol_vec_loc,2)
410 
411  ! initialize ierr
412  ierr = 0
413 
414  ! operations depending on X style
415  select case (x_style)
416  case (1) ! prescribed
417  ! test allocation
418  if (allocated(sol_vec_tot)) then
419  if (size(sol_vec_tot,1).ne.size(sol_vec_loc,1) .or. &
420  &size(sol_vec_tot,2).ne.size(sol_vec_loc,2)) then
421  ierr = 1
422  err_msg = 'Local and total solution vectors need to &
423  &be of the same length'
424  chckerr(err_msg)
425  end if
426  else
427  allocate(sol_vec_tot(size(sol_vec_loc,1),n_r))
428  end if
429  sol_vec_tot = sol_vec_loc
430  case (2) ! fast
431  ! set local and total nr. of modes
432  n_mod_loc = n_mod_x
433  n_mod_tot = size(sec_x_ind,2)
434 
435  ! test allocation
436  if (allocated(sol_vec_tot)) then
437  if (size(sol_vec_tot,1).ne.n_mod_tot .or. &
438  &size(sol_vec_tot,2).ne.size(sol_vec_loc,2)) then
439  ierr = 1
440  err_msg = 'Total solution vectors needs to be of &
441  &correct length '//trim(i2str(n_mod_tot))
442  chckerr(err_msg)
443  end if
444  else
445  allocate(sol_vec_tot(n_mod_tot,n_r))
446  end if
447 
448  ! track all possible modes and copy
449  sol_vec_tot = 0._dp
450  do ld = 1,n_mod_tot
451  do kd = 1,n_r
452  kd_loc = kd + i_min - 1
453  if (sec_x_ind(kd_loc,ld).ne.0) sol_vec_tot(ld,kd) = &
454  &sol_vec_loc(sec_x_ind(kd_loc,ld),kd)
455  end do
456  end do
457  end select
458  end function calc_tot_sol_vec
