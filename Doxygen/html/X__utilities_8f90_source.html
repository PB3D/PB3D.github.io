59  ! Gets one of the the local ranges of contiguous tensorial perturbation
60  ! variables to be printed or read during one call of the corresponding HDF5
61  ! variables.
62  ! More specifically, a range of indices k in the first dimension is given
63  ! for every value of the indices m in the second dimension. An example is
64  ! now given for the subrange [2:3,2:5] of a the total range [1:5, 1:5]. For
65  ! asymmetric variables the situation is simple: The k range is [2:3] for all
66  ! 5 values of m. However, for symmetric variables, the upper diagonal values
67  ! are not stored, which gives k ranges [2:3], [3:3] and no range for m = 4
68  ! and 5.
69  ! This routine then translates these ranges to the corresponding 1-D ranges
70  ! that are used in the actual variables. For above example, the total
71  ! indices are
72  ! [ 1 6 11 16 21 ]
73  ! [ 2 7 12 17 22 ]
74  ! [ 3 8 13 18 23 ] -> [7:8], [12:13], [17:18] and [22:23],
75  ! [ 4 9 14 19 24 ]
76  ! [ 5 10 15 20 25 ]
77  ! for asymmetric variables and
78  ! [ 1 2 3 4 5 ]
79  ! [ 2 6 7 8 9 ]
80  ! [ 3 7 10 11 12 ] -> [6:7], [10:10], [:] and [:],
81  ! [ 4 8 11 13 14 ]
82  ! [ 5 9 12 14 15 ]
83  ! for symmetric variables.
84  ! These can then related to the local indices for the variables in this
85  ! perturbation job. For above example, the results are:
86  ! [1:2], [3:4], [5:6] and [7:8],
87  ! for asymmetric variables and
88  ! [1:2], [3:3], [:] and [:],
89  ! for symmetric variables.
90  ! As can be seen, the local ranges of the variables in the submatrix of this
91  ! perturbation job are (designed to be) contiguous, but the total ranges of
92  ! the variables in the submatrix are clearly not in general.
93  ! The procedure outputs both the local and total ranges.
94  subroutine get_sec_x_range(sec_X_range_loc,sec_X_range_tot,m,sym,lim_sec_X)
95  use x_vars, only: n_mod_x
96  use num_utilities, only: c

98  ! input / output
99  integer, intent(inout) :: sec_X_range_loc(2) ! start and end of local range in dimension 1 (vertical)
100  integer, intent(inout) :: sec_X_range_tot(2) ! start and end of total range in dimension 1 (vertical)
101  integer, intent(in) :: m ! dimension 2 (horizontal)
102  logical, intent(in) :: sym ! whether the variable is symmetric
103  integer, intent(in), optional :: lim_sec_X(2,2) ! limits of m_X (pol. flux) or n_X (tor. flux)
104 
105  ! local variables
106  integer :: k_range_loc(2) ! local range of k
107  integer :: k ! counter
108  integer :: n_mod ! local n_mod
109 
110  ! set number of modes of dimension 1
111  if (present(lim_sec_x)) then
112  n_mod = lim_sec_x(2,1)-lim_sec_x(1,1)+1
113  else
114  n_mod = n_mod_x
115  endif
116 
117  ! initialize secondary X range
118  k_range_loc = [n_mod+1,0] ! initialize to inverted values, out of bounds
119 
120  ! find start and end of k range
121  find_start: do k = 1,n_mod
122  if (is_necessary_x(sym,[k,m],lim_sec_x)) then ! first necessary pair [k,m]
123  k_range_loc(1) = k
124  exit find_start
125  end if
126  end do find_start
127  find_stop: do k = n_mod,k_range_loc(1),-1
128  if (is_necessary_x(sym,[k,m],lim_sec_x)) then ! last necessary pair [k,m]
129  k_range_loc(2) = k
130  exit find_stop
131  end if
132  end do find_stop
133 
134  ! translate k range and m value to 1-D index
135  do k = 1,2
136  sec_x_range_loc(k) = c([k_range_loc(k),m],sym,n_mod_x,lim_sec_x)
137  sec_x_range_tot(k) = c(sec_ind_loc2tot(k_range_loc(k),m,lim_sec_x),&
138  &sym,n_mod_x)
139  end do
140  end subroutine get_sec_x_range

183  ! Calculate memory in MB necessary for X variables of a certain order
184  ! - order 1: 4x n_par_X x n_geo x loc_n_r x n_mod
185  ! - order 2: 2x n_par_X x n_geo x loc_n_r x n_mod^2
186  ! 4x n_par_X x n_geo x loc_n_r x n_mod(n_mod+1)/2
187  ! - higher order: not used
188  ! where n_par_X x n_geo x loc_n_r should be passed as 'arr_size' and n_mod
189  ! as well
190  integer function calc_memory_x(ord,arr_size,n_mod,mem_size) result(ierr)
191  use iso_c_binding
192 
193  character(*), parameter :: rout_name = 'calc_memory_X'
194 
195  ! input / output
196  integer, intent(in) :: ord ! order of data
197  integer, intent(in) :: arr_size ! size of part of X array
198  integer, intent(in) :: n_mod ! number of modes
199  real(dp), intent(inout) :: mem_size ! total size
200 
201  ! local variables
202  integer(C_SIZE_T) :: dp_size ! size of dp
203  character(len=max_str_ln) :: err_msg ! error message
204 
205  ! initialize ierr
206  ierr = 0
207 
208  call lvl_ud(1)
209 
210  ! get size of complex variable
211  dp_size = 2*sizeof(1._dp) ! complex variable
212 
213  ! calculate memory depending on order
214  select case(ord)
215  case (1) ! vectorial data: U, DU
216  ! set memory size
217  mem_size = 4*arr_size
218  mem_size = mem_size*n_mod**ord
219  mem_size = mem_size*dp_size
220  case (2) ! tensorial data: PV, KV
221  ! set memory size
222  mem_size = arr_size
223  mem_size = mem_size*(2*n_mod**ord+4*n_mod*(n_mod+1)/2)
224  mem_size = mem_size*dp_size
225  case default
226  ierr = 1
227  err_msg = 'Orders > 2 are not implemented'
228  chckerr(err_msg)
229  end select
230 
231  ! convert B to MB
232  mem_size = mem_size*1.e-6_dp
233 
234  ! apply 50% safety factor (empirical)
235  mem_size = mem_size*1.5_dp
236 
237  ! test overflow
238  if (mem_size.lt.0) then
239  ierr = 1
240  chckerr('Overflow occured')
241  end if
242 
243  call lvl_ud(-1)
244  end function calc_memory_x
