1 !------------------------------------------------------------------------------!
2 ! Utilities pertaining to HDF5 and XDMF !
3 ! Notes about XDMF: !
4 ! - Collections can be spatial or temporal. If a variable is to be !
5 ! evolved in time or if its domain is decomposed (e.g. the same !
6 ! physical variable is defined in multiple HDF5 variables), then the !
7 ! attribute name of this variable has to be the same for all the grids !
8 ! in the collection. !
9 ! - If no collection is used, but just multiple grids, the attribute !
10 ! can be different but does not have to be, as the different grid are !
11 ! distinguished by their different grid names, not by the attribute !
12 ! of the variables they contain. !
13 ! - The selection of hyperslabs is used, as described here: !
14 ! http://davis.lbl.gov/Manuals/HDF5-1.8.7/UG/12_Dataspaces.html !
15 ! - Chunking is used for partial I/O, as described here: !
16 ! https://www.hdfgroup.org/HDF5/doc/Advanced/Chunking/ !
17 ! As no chunk cache is reused, the w0 setting should be 1. !
18 ! Furthermore, the number of slots is chosen to be equal to the number !
19 ! of chunks. And Finally, the chunk size is chosen to be the size of !
20 ! the previous dimensions, if it does not exceed 4GB !
21 !------------------------------------------------------------------------------!

44  ! Sets up the chunk property list and/or the 1D hyperslabs that correspond
45  ! to a local subset of lim_tot given by the limits lim_loc. An example for
46  ! the hyperslab is given in 3 dimensions with size n = 5,3,3:
47  ! for dim = 1, limits: 2..4
48  ! for dim = 2, limits: 3..3
49  ! for dim = 3, limits: 2..3.
50  ! The 1D equivalent can be represented as:
51  ! [.....][.....][.....] | [.....][.....][.....] | [.....][.....][.....]
52  ! dimension 1 will only allow for the following elements (given by "-"):
53  ! [.---.][.---.][.---.] | [.---.][.---.][.---.] | [.---.][.---.][.---.]
54  ! dimension 2 will only allow for the following elements:
55  ! [.....][.....][-----] | [.....][.....][-----] | [.....][.....][-----]
56  ! dimension 3 will only allow for the following elements:
57  ! [.....][.....][.....] | [-----][-----][-----] | [-----][-----][-----]
58  ! so that the total is given by:
59  ! [.....][.....][.....] | [.....][.....][.....] | [.....][.....][.---.]
60  ! In practice, it is easiest to work with a full selection, where for every
61  ! dimension the ranges that do not correspond to the range of that dimension
62  ! are taken away. Clearly, if a dimension has the full range, nothing will
63  ! be taken out:
64  ! block_i = n_1*n_2*..*n_(i-1) * (b_i-a_i+1) ,
65  ! stride_i = n_1*n_2*..*n_(i-1) * (B_i-A_i+1) ,
66  ! offset_i = n_1*n_2*..*n_(i-1) * (a_i-A_i) ,
67  ! count_i = n_(i+1)*n_(i+2)*..*n_N ,
68  ! where a_i and b_i represent the local limits of dimension i, A_i and B_i
69  ! the total ones and the number of dimensions is N, as can be verified.
70  ! The chunk property list for creation can be set up so that its size is
71  ! equal to the largest dimensions of full range, or an integer part of that
72  ! if it exceeds 4GB. Since the variables don't need to be used more than
73  ! once, either nbytes or nslots could be set to 0 in an access property
74  ! list, but this is currently not done.
75  integer function set_1d_vars(lim_tot,lim_loc,space_id,c_plist_id) &
76  &result(ierr)
77 #if ldebug
78  use num_vars, only: rank
79 #endif
80 
81  character(*), parameter :: rout_name = 'set_1D_vars'

83  ! input / output
84  integer, intent(in) :: lim_tot(:,:) ! total limits
85  integer, intent(in) :: lim_loc(:,:) ! local limits
86  integer(HID_T), intent(in), optional :: space_id ! dataspace identifier
87  integer(HID_T), intent(inout), optional :: c_plist_id ! chunk creation property list identifier
88 
89  ! local variables
90  integer :: id, kd ! counters
91  integer :: n_dims ! nr. of dimensions
92  integer :: n_prod(2) ! n_prod(1) = n_1*n_2*..*n_(i-1), n_prod(2) = n_prod(1)*n_i
93  integer :: div_dim ! first divided dimension
94  integer :: chunk_size ! size of chunk
95  integer(HSIZE_T) :: block(1) ! block size in memory
96  integer(HSIZE_T) :: stride(1) ! stride in memory
97  integer(HSIZE_T) :: count(1) ! nr. of repetitions of block in memory
98  integer(HSIZE_T) :: offset(1) ! offset in memory
99  integer(HSIZE_T) :: chunk_dims(1) ! chunk dimensions
100  real(dp) :: max_chunk_size ! maximum 4 GB (from manual)
101  real(dp) :: min_chunk_size ! minimum 10 MB (from experience)
102  character(len=max_str_ln) :: err_msg ! error message
103 #if ldebug
104  integer :: istat ! status
105 #endif
106 
107  ! initialize ierr
108  ierr = 0
109 
110  ! set n_dims
111  n_dims = size(lim_tot,1)
112 
113 #if ldebug
114  if (size(lim_loc,1).ne.n_dims) then
115  ierr = 1
116  err_msg = 'lim_tot and lim_loc are not compatible'
117  chckerr(err_msg)
118  end if
119  if (size(lim_tot,2).ne.2 .or. size(lim_loc,2).ne.2) then
120  ierr = 1
121  err_msg = 'lim_tot and_lim_loc need to contain 2 columns'
122  chckerr(err_msg)
123  end if
124  do id = 1,n_dims
125  if (lim_tot(id,1).gt.lim_loc(id,1) .or. &
126  &lim_tot(id,2).lt.lim_loc(id,2)) then
127  write(*,*) rank, 'lim_tot = ', lim_tot
128  write(*,*) rank, 'lim_loc = ', lim_loc
129  ierr = 1
130  err_msg = 'Total limits must comprise local ones'
131  chckerr(err_msg)
132  end if
133  end do
134 
135  if (debug_set_1d_vars) then
136  write(*,*,iostat=istat) rank, 'in set_1D_vars:'
137  write(*,*,iostat=istat) rank, 'lim_tot_lo = ', lim_tot(:,1)
138  write(*,*,iostat=istat) rank, 'lim_tot_hi = ', lim_tot(:,2)
139  write(*,*,iostat=istat) rank, 'lim_loc_lo = ', lim_loc(:,1)
140  write(*,*,iostat=istat) rank, 'lim_loc_hi = ', lim_loc(:,2)
141  end if
142 #endif
143 
144  ! initialize divided dimension
145  div_dim = n_dims+1
146 
147  ! set hyperslab to be everything if requested
148  if (present(space_id)) then
149  call h5sselect_all_f(space_id,ierr)
150  chckerr('Failed to select hyperslab')
151  end if
152 
153  ! loop over dimensions to possibly restrict hyperslab and set div_dim
154  do id = 1,n_dims
155  ! only restrict hyperslab if local range differ from total one
156  if (lim_loc(id,1).gt.lim_tot(id,1) .or. &
157  &lim_loc(id,2).lt.lim_tot(id,2)) then
158  ! set divided dimension if first time
159  if (div_dim.eq.n_dims+1) div_dim = id
160 
161  ! calculations if hyperslab restriction requested
162  if (present(space_id)) then
163  ! set auxiliary variables
164  n_prod(1) = product(lim_tot(1:id-1,2)-lim_tot(1:id-1,1)+1)
165  n_prod(2) = n_prod(1)*(lim_tot(id,2)-lim_tot(id,1)+1)
166 
167 #if ldebug
168  if (debug_set_1d_vars) then
169  write(*,*,iostat=istat) rank, 'dimension', id, 'of', &
170  &n_dims
171  write(*,*,iostat=istat) rank, 'n_prod = ', n_prod
172  end if
173 #endif
174 
175  block = (lim_loc(id,2)-lim_loc(id,1)+1) * n_prod(1)
176  stride = n_prod(2)
177  offset = (lim_loc(id,1)-lim_tot(id,1)) * n_prod(1)
178  count = 1
179  do kd = id+1,n_dims
180  count = count*(lim_tot(kd,2)-lim_tot(kd,1)+1)
181  end do
182 
183 #if ldebug
184  if (debug_set_1d_vars) then
185  write(*,*,iostat=istat) rank, 'block, offset, stride, &
186  &count = ', block, offset, stride, count
187  end if
188 #endif
189 
190  call h5sselect_hyperslab_f(space_id,h5s_select_and_f,&
191  &offset,count,ierr,stride=stride,block=block)
192  chckerr('Failed to select hyperslab')
193  end if
194 #if ldebug
195  if (debug_set_1d_vars) then
196  write(*,*,iostat=istat) rank, 'total range DIFFERS from &
197  &local'
198  end if
199  else
200  if (debug_set_1d_vars) then
201  write(*,*,iostat=istat) rank, 'total range EQUAL to local'
202  end if
203 #endif
204  end if
205  end do
206 
207  ! set chunk property list if requested
208  max_chunk_size = 4.e9_dp / sizeof(1._dp) ! maximum 4 GB (from manual)
209  min_chunk_size = min(10.e6_dp/sizeof(1._dp),&
210  &product(lim_tot(:,2)-lim_tot(:,1)+1._dp)) ! maximum 10 MB (from experience), or limit to total variable size
211  if (present(c_plist_id)) then
212  ! set up variables
213  chunk_size = 1
214  id = 1
215  do while (chunk_size.le.min_chunk_size .and. id.le.size(lim_tot,1))
216  chunk_size = chunk_size * (lim_tot(id,2)-lim_tot(id,1)+1)
217  id = id + 1
218  end do
219 #if ldebug
220  if (debug_set_1d_vars) then
221  write(*,*,iostat=istat) rank, 'suggested chunk size', chunk_size
222  end if
223 #endif
224  chunk_size = chunk_size / (1 + floor(chunk_size/max_chunk_size)) ! limit to max chunk size
225 #if ldebug
226  if (debug_set_1d_vars) then
227  write(*,*,iostat=istat) rank, 'def. chunk size', chunk_size
228  end if
229 #endif
230 
231  ! set creation property list if provided
232  if (present(c_plist_id)) then
233  chunk_dims = chunk_size
234  call h5pcreate_f(h5p_dataset_create_f,c_plist_id,ierr)
235  chckerr('Failed to create property list')
236  call h5pset_chunk_f(c_plist_id,1,chunk_dims,ierr) ! one dimension for 1D chunk
237  chckerr('Failed to create chunk')
238  end if
239  end if
240  end function set_1d_vars

242  ! Probe HDF5 file for group existence.
243  integer function probe_hdf5_group(HDF5_name,group_name,group_exists) &
244  &result(ierr)
245  use mpi_vars, only: hdf5_lock
246  use mpi_utilities, only: lock_req_acc, lock_return_acc
247 
248  character(*), parameter :: rout_name = 'probe_HDF5_group'
249 
250  ! input / output
251  character(len=*), intent(in) :: HDF5_name ! name of HDF5 file
252  character(len=*), intent(in) :: group_name ! name of group to probe for
253  logical, intent(inout) :: group_exists ! whether group exists
254 
255  ! local variables
256  integer :: istat ! status
257  integer(HID_T) :: HDF5_i ! file identifier
258  integer(HID_T) :: group_id ! group identifier
259 
260  ! initialize ierr
261  ierr = 0
262 
263  ! initialize FORTRAN predefined datatypes
264  call h5open_f(ierr)
265  chckerr('Failed to initialize HDF5')
266 
267  ! wait for file access in a non-blocking way
268  ierr = lock_req_acc(hdf5_lock,blocking=.false.)
269  chckerr('')
270 
271  ! open the file
272  call h5fopen_f(hdf5_name,h5f_acc_rdonly_f,hdf5_i,ierr)
273  chckerr('Failed to open file. Does it exist?')
274 
275  ! disable error messages
276  call h5eset_auto_f(0,ierr)
277  chckerr('Failed to disable error printing')
278 
279  ! try to open group
280  call h5gopen_f(hdf5_i,group_name,group_id,istat)
281  group_exists = istat.eq.0
282 
283  ! reenable error messages
284  call h5eset_auto_f(1,ierr)
285  chckerr('Failed to enable error printing')
286 
287  ! close group
288  if (group_exists) then
289  call h5gclose_f(group_id,ierr)
290  chckerr('Failed to close head group')
291  end if
292 
293  ! close the HDF5 file
294  call h5fclose_f(hdf5_i,ierr)
295  chckerr('failed to close HDF5 file')
296 
297  ! return lock
298  ierr = lock_return_acc(hdf5_lock)
299  chckerr('')
300 
301  ! close FORTRAN interfaces and HDF5 library.
302  call h5close_f(ierr)
303  chckerr('Failed to close FORTRAN HDF5 interface')
304  end function probe_hdf5_group
