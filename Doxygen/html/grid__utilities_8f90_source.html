970  ! Calculates volume integral on a 3D grid.
971  ! Two angular and one normal variable has to be provided on a the grid. If
972  ! the i'th dimension of the grid is equal to one, the function to be
973  ! integrated is assumed not to vary in this dimension.
974  ! Furthermore, if i is 1 or 2, the corresponding i'th (angular) variable is
975  ! the only variable that is assumed to vary in that dimension. The other
976  ! angular variable as well as the normal variable are assumed to be constant
977  ! like the function itself. However, if i is 3, an error is displayed as
978  ! this does not represent a physical situation.
979  ! A common case through which to understand this is the axisymmetric case
980  ! where the first angular variable theta varies in the dimensions 1 and 3,
981  ! the second angular variable zeta varies only in the dimension 2, and the
982  ! normal variable only varies in the dimension 3.
983  ! Alternatively, there is the case of a grid-aligned set of coordinates
984  ! theta and alpha, where the first dimension corresponds to the direction
985  ! along the magnetic field line, the second to the geodesical direction and
986  ! the third to the normal direction. If the calculations for different field
987  ! lines are decoupled, the variation in the second dimension is not taken
988  ! into account and no integration happens along it.
989  ! Internally, the angular variables and the normal variable are related to
990  ! the coordinates (x,y,z) that correspond to the three dimensions. They thus
991  ! form a computational orthogonal grid to which the original coordinates are
992  ! related through the transformation of Jacobians:
993  ! Jxyz = J r_F_z (ang_1_x ang_2_y-ang_1_y ang_2_x) ,
994  ! so that the integral becomes
995  ! sum_xyz f(x,y,z) J(x,y,z) r_F_z (ang_1_x ang_2_y-ang_1_y ang_2_x) dxdydz
996  ! where dx, dy and dz are all trivially equal to 1.
997  ! The integrand has to be evaluated at the intermediate positions inside the
998  ! cells. This is done by taking the average of the 2^3=8 points for fJ as
999  ! well as the transformation of the Jacobian.
1000  ! Note: if the coordinates are independent, this method is equivalent to the
1001  ! repeated numerical integration using the trapezoidal method, NOT Simpson's
1002  ! 3/8 rule!
1003  ! Note: by setting debug_calc_int_vol, this method can be compared to the
1004  ! trapezoidal and simple method for independent coordinates, again NOT for
1005  ! Simpson's 3/8 rule!
1006  ! The Simpson's 3/8 rule could be developed but it is not of great
1007  ! importance.
1008  integer function calc_int_vol(ang_1,ang_2,norm,J,f,f_int) result(ierr)
1009 #if ldebug
1010  use num_vars, only: rank, n_procs
1011 #endif
1012 
1013  character(*), parameter :: rout_name = 'calc_int_vol'
1014 
1015  ! input / output
1016  real(dp), intent(in) :: ang_1(:,:,:), ang_2(:,:,:), norm(:) ! coordinate variables

1299  ! Set up the factors for the derivative calculation in a matrix "A" using
1300  ! finite differences of order "ord" with precision "prec", by which is meant
1301  ! that the order of the error is at least ~ Delta^(prec+1).
1302  ! Afterwards, the matrix A has to be multiplied with the variable to be
1303  ! derived to obtain the requested derivative.
1304  ! For a particular order and precision, the number of different points that
1305  ! have to be combined is ord+prec, however, as it is better to use symmetric
1306  ! expressions, this number is possibly aumented by one. This number is saved
1307  ! in "n_loc".
1308  ! The index kd is then defined to go from -(n_loc-1)/2 .. (n_loc-1)/2, which
1309  ! is translated to local coordinates by adding (n_loc+1)/2 and to total
1310  ! coordinates by adding generally the index in total coordinates i where the
1311  ! local problem is to be set up, but capping it by (n_loc+1)/2 and
1312  ! n-(n_loc-1)/2, so by using
1313  ! k_tot = kd + max((n_loc+1)/2,min(id,n-(n_loc-1)/2)),
1314  ! so that these never go out of bounds 1 .. n.
1315  ! The local matrix element "mat_loc" is then set up as follows:
1316  ! [ 1 1 ... 1 1 ]
1317  ! [ D^i-2_i D^i-1,i 0 D^i+1_i D^i+2_i ]
1318  ! [ (D^i-2_i)^2/2! (D^i-1,i)^2/2! 0 (D^i+1_i)^2/2! (D^i+2_i)^2/2! ]
1319  ! [ (D^i-2_i)^3/3! (D^i-1,i)^3/3! 0 (D^i+1_i)^3/3! (D^i+2_i)^3/3! ]
1320  ! [ ... ... ... ... ... ]
1321  ! for bulk matrices ((n_loc+1)/2 <= i <= n-(n_loc-1)/2), with
1322  ! D^j_i = x(j)-x(i).
1323  ! For other elements, this is shifted, e.g.:
1324  ! mat_loc(j,k) = (x(k_tot)-X(i))^(j-1)
1325  ! where j = 1..n_loc and k = -(n_mod-1)/2..(n_mod-1)/2. k_tot is defined
1326  ! above.
1327  ! The solution A_i of mat_loc A_i = rhs_loc is then saved in an individual
1328  ! row i in the total matrix A, where rhs_loc = [0,0,..,0,1,0,..,0] with the
1329  ! unit indicating the order of the derivative.
1330  ! However, instead of saving the entire matrix A, only the elements from A_i
1331  ! are saved, omitting the zero's.
1332  ! For equidistant grids, the situation becomes easier as D^j_i is not a
1333  ! function of id and the rows of local matrix are displaced copies while the
1334  ! first rows are the mirror image of the last rows so that the local
1335  ! matrices become symmetric. Also, the size of the total variables has to be
1336  ! passed in n, in contrast to the regular version, where it is set
1337  ! automatically.

1492  &mat_loc(2,kd+(n_loc+1)/2)**(jd-1)

1494  end do

1503  chckerr(err_msg)

1514  ! Set up the factors for the interpolation calculation in a matrix "A" using
1515  ! a polynomial order "ord".
1516  ! This is done using Barycentric Lagrangian polynomials, which are a
1517  ! computationally more advantageous way of expressing the Lagrangian
1518  ! interpolating polynomial:
1519  ! L(x) = sum_j=0^N (w_j f_j /(x-x_j)) / sum_j=0^N (w_j/(x-x_j)) , with
1520  ! w_j = 1 / l'(x_j) , where
1521  ! l(x) = product_i=0^N (x-x_i) , so that
1522  ! l'(x_j) = product_j.ne.i (x_j-x_i)
1523  ! [https://people.maths.ox.ac.uk/trefethen/barycentric.pdf, 02/05/2016]
1524  ! The optimal range of grid points "x" around the interpolation point
1525  ! "x_interp" is first calculated, after which the weight functions w_j are
1526  ! calculated for each of the interpolation points, multiplied by 1 /
1527  ! (x-x_j). The result is scaled by the sum of all, and this is tabulated.
1528  ! Note: To avoid numerical problems, the factors (x_j-x_i) are divided by a
1529  ! characteristic length len, equal to the average length of the working
1530  ! interval. This scales both the enumerator and the denominator by len^N,
1531  ! which therefore cancel. Also, if the interpolation point x_interp is
1532  ! within a tiny tolerance of the grid points x, the machinery is bypassed to
1533  ! avoid unstabilities.
1534  ! Optionally, this routine can do trigonometric interpolation instead of
1535  ! polynomial, which results in using sin((x-x_j)/2) instead of (x-x_j).
1536  ! [https://en.wikipedia.org/wiki/Trigonometric_interpolation, 01/09/2016]
1537  ! Currently, this only works for an odd number of points, as there is also
1538  ! an odd number of constraints.
1539  ! Note that this can be used to resample a periodic function that has been
1540  ! sampled on an irregular grid, in order to calculate the fourier transform
1541  ! with a FFT, which requires a regular grid. In this case, a full 2pi period
1542  ! is assumed.
1543  ! The fact that this is can be done can be seen by realizing that
1544  ! trigonometric interpolation can also be written in Lagrange form, and
1545  ! subsequently in barycentric Lagrange form. There is an additional factor
1546  ! 1/2 in l'(x_j), due to the derivation of sin((x-x_k)/2) but this cancels
1547  ! as it is present in both enumerator and denominator.
1548  ! Optionally also a normalization length can be provided so that the
1549  ! individual terms in the interpolation do not blow up. This can be useful
1550  ! for example for interpolation with many points.
1551  integer function setup_interp_data(x,x_interp,A,ord,is_trigon,norm_len) &
1552  &result(ierr)
1553  use grid_vars, only: disc_type
1554  use num_utilities, only: con2dis
1555 
1556  character(*), parameter :: rout_name = 'setup_interp_data'
1557 
1558  ! input / output
1559  real(dp), intent(in) :: x(:) ! independent variable of tabulated magnitude
1560  real(dp), intent(in) :: x_interp(:) ! values of x at which to interpolate
1561  type(disc_type), intent(inout) :: A ! interpolation mat
1562  integer, intent(in) :: ord ! order of derivative
1563  logical, intent(in), optional :: is_trigon ! trigonometric interpolation
1564  real(dp), intent(in), optional :: norm_len ! custom normalization length
1565 
1566  ! local variables
1567  integer :: id, jd, kd ! counters
1568  integer :: n ! size of x_interp
1569  integer :: n_loc ! nr. of points used for interpolation locally
1570  real(dp) :: len ! interval length
1571  real(dp) :: x_interp_disc ! discrete index of current x_interp
1572  real(dp), parameter :: tol = 1.0e-8 ! tolerance
1573  character(len=max_str_ln) :: err_msg ! error message
1574  real(dp), allocatable :: weight(:) ! weights w_j
1575  real(dp) :: weight_loc ! local weight factor
1576  logical :: is_trigon_loc ! local is_trigon
1577 
1578  ! initialize ierr
1579  ierr = 0
1580 
1581  ! set local is_trigon
1582  is_trigon_loc = .false.
1583  if (present(is_trigon)) is_trigon_loc = is_trigon
1584 
1585  ! tests
1586  if (ord.lt.1) then
1587  ierr = 1
1588  err_msg = 'order has to be at least 1'
1589  chckerr(err_msg)
1590  else if (ord+1.gt.size(x)) then
1591  ierr = 1
1592  err_msg = 'order can be at most size(x)-1 = '//&
1593  &trim(i2str(size(x)-1))
1594  chckerr(err_msg)
1595  end if
1596  if (is_trigon_loc .and. mod(ord,2).ne.0) then
1597  ierr = 1
1598  err_msg = 'Need an even order, so that there is an odd number of &
1599  &points'
1600  chckerr(err_msg)
1601  end if
1602 
1603  ! set n_loc and n
1604  n_loc = ord+1
1605  n = size(x_interp)
1606 
1607  ! set variables
1608  allocate(weight(n_loc))
1609  ierr = a%init(n,n_loc)
1610  chckerr('')
1611 
1612  ! iterate over all x_interp values
1613  do id = 1,n
1614  ! find the base of the interpolation value
1615  ierr = con2dis(x_interp(id),x_interp_disc,x)
1616  chckerr('')
1617 
1618  ! set up start id
1619  if (n_loc.lt.size(x)) then
1620  a%id_start(id) = ceiling(x_interp_disc-n_loc*0.5_dp) ! get minimum of bounding indices
1621  a%id_start(id) = max(1,min(a%id_start(id),size(x)-n_loc+1)) ! limit to lie within x
1622  else
1623  a%id_start(id) = 1 ! all values of x are used
1624  end if
1625 
1626  ! check for (near) exact match
1627  if (mod(x_interp_disc,1._dp).lt.tol) then
1628  ! directly set the correct index to 1
1629  a%dat(id,:) = 0._dp
1630  a%dat(id,nint(x_interp_disc)-a%id_start(id)+1) = 1._dp
1631  else
1632  ! calculate w_j/(x-x_j)
1633  len = (x(a%id_start(id)+n_loc-1) - x(a%id_start(id)))/n_loc
1634  if (is_trigon_loc) len = sin(len/2) ! take sines
1635  if (present(norm_len)) len = norm_len
1636 
1637  ! set up the interp. coeffs. due to each of the points used
1638  weight = 1._dp
1639  do jd = 1,n_loc
1640  do kd = 1,n_loc
1641  ! basis of local weight
1642  if (kd.ne.jd) then ! skip k = j
1643  ! calculate l'(x_j) = product_k.ne.j (x_j-x_k)
1644  weight_loc = &
1645  &x(a%id_start(id)-1+jd)-x(a%id_start(id)-1+kd)
1646  else
1647  ! multiply additionally by (x_interp - x_j)
1648  weight_loc = x_interp(id)-x(a%id_start(id)-1+jd)
1649  end if
1650 
1651  ! modification of local weight
1652  if (is_trigon_loc) weight_loc = sin(weight_loc/2) ! take sines
1653  weight_loc = weight_loc/len ! normalize
1654 
1655  ! including modified local weight
1656  weight(jd) = weight(jd) * weight_loc
1657  end do
1658  end do
1659 
1660  ! invert
1661  weight = 1._dp/weight
1662 
1663  ! scale by sum and save in A
1664  a%dat(id,:) = weight/sum(weight)
1665  end if
1666  end do
1667  end function setup_interp_data

1935  ! Trim a grid, removing any overlap between the different regions.
1936  ! by default, the routine assumes a symmetric ghost region and cuts as many
1937  ! grid points from the end of the previous process as from the beginning of
1938  ! the next process, but if the number of overlapping grid points is odd, the
1939  ! next process looses one more point.
1940  ! optionally, the trimmed indices in the normal direction can be provided in
1941  ! "norm_id", i.e. the indices in the old, untrimmed grid that correspond to
1942  ! the start and end indices of the trimmed grid. E.g. if
1943  ! - proc 0: 3 ... 25
1944  ! - proc 1: 20 ... 50
1945  ! then the trimmed grid will be:
1946  ! - proc 0: 3 ... 22
1947  ! - proc 1: 23 ... 50
1948  ! which is shifted down by 2 to
1949  ! - proc 0: 1 ... 20
1950  ! - proc 1: 21 ... 48
1951  ! in the trimmed grid. The indices of the previous step (3 & 22 and 23 & 50)
1952  ! are saved in norm_id.
1953  integer function trim_grid(grid_in,grid_out,norm_id) result(ierr)
1954  use num_vars, only: n_procs, rank
1955  use mpi_utilities, only: get_ser_var
1956 
1957  character(*), parameter :: rout_name = 'trim_grid'
1958

1960  type(grid_type), intent(in) :: grid_in ! input grid
1961  type(grid_type), intent(inout) :: grid_out ! trimmed grid
1962  integer, intent(inout), optional :: norm_id(2) ! normal indices corresponding to trimmed part
1963 
1964  ! local variables
1965  integer, allocatable :: tot_i_min(:) ! i_min of grid of all processes
1966  integer, allocatable :: tot_i_max(:) ! i_max of grid of all processes
1967  integer :: i_lim_out(2) ! i_lim of output grid
1968  integer :: n_out(3) ! n of output grid
1969 
1970  ! initialize ierr
1971  ierr = 0
1972

1974  if (grid_in%divided) then
1975  ! get min_i's of the grid_in
1976  ierr = get_ser_var([grid_in%i_min],tot_i_min,scatter=.true.)

1980  ierr = get_ser_var([grid_in%i_max],tot_i_max,scatter=.true.)
1981  chckerr('')

1988  else

1990  end if
1991  if (rank.lt.n_procs-1) then
1992  i_lim_out(2) = min(tot_i_max(n_procs),tot_i_max(rank+1)-&
1993  &floor((tot_i_max(rank+1)-tot_i_min(rank+2)+1._dp)/2))
1994  else
1995  i_lim_out(2) = tot_i_max(n_procs)
1996  end if

1999  i_lim_out(1) = max(min(i_lim_out(1),grid_in%i_max),grid_in%i_min)
2000  i_lim_out(2) = max(min(i_lim_out(2),grid_in%i_max),grid_in%i_min)
2001 
2002  ! get min_i's and max_i's of the grid_out, not shifted by min of
2003  ! first process
2004  ierr = get_ser_var([i_lim_out(1)],tot_i_min,scatter=.true.)
2005  chckerr('')
