421 !> Reconstructs grid variables from PB3D HDF5 output.
422 !!
423 !! Also, if \c rich_lvl is provided, <tt>_R_[rich_lvl]</tt> is appended to
424 !! the data name if it is > 0.
425 !!
426 !! With \c tot_rich the information from previous Richardson levels can be
427 !! combined.
428 !!
429 !! Furthermore, using \c lim_pos, you can obtain a subset of the data by
430 !! directly passing its limits to the underlying HDF5 routine. This refers
431 !! to the position dimensions only. If provided, the normal limits of a
432 !! divided grid refer to the subset, as in copy_grid().
433 !!
434 !! \note
435 !! -# <tt>grid_</tt> is added in front the data_name.
436 !! -# By providing \c lim_pos equal to 0 in the angular dimensions, the
437 !! angular part can be discarded when reconstructing the grid.

439 !! \return ierr

452 integer, intent(in), optional :: rich_lvl !< Richardson level to reconstruct
453 logical, intent(in), optional :: tot_rich !< whether to combine with previous Richardson levels
454 integer, intent(in), optional :: lim_pos(3,2) !< position limits of subset of data
455 integer, intent(in), optional :: grid_limits(2) !< i_limit of grid

482 ! possibly change n to user-specified
483 if (present(lim_pos)) then
484 if (lim_pos(1,1).ge.0 .and. lim_pos(1,2).ge.0) n(1) = &
485 &lim_pos(1,2)-lim_pos(1,1)+1
486 if (lim_pos(2,1).ge.0 .and. lim_pos(2,2).ge.0) n(2) = &
487 &lim_pos(2,2)-lim_pos(2,1)+1
